  <script>
    function checkAnswers() {

      const answers = {
        q1: "7:00",
        q2: "face",
        q3: "breakfast",
        q4: "8:00",
        q5: "math",
        q6: "2:00",
        q7: "football",
        q8: "homework"
      };

      let score = 0;
      let total = 8;

      for (let key in answers) {
        let userAnswer = document.getElementById(key).value.trim().toLowerCase();
        let correctAnswer = answers[key].toLowerCase();

        if (userAnswer === correctAnswer) {
          score++;
          document.getElementById(key).style.borderColor = "green";
        } else {
          document.getElementById(key).style.borderColor = "red";
        }
      }

      const result = document.getElementById("result");

      result.innerHTML = `You got ${score} out of ${total} correct!`;

      if (score === total) {
        result.className = "result correct";
      } else {
        result.className = "result wrong";
      }
    }
  </script>
